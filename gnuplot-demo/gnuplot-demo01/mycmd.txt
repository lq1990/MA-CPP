# gnuplot cli in a file
reset

# set term postscript eps enhanced
# set output 'myp.eps'

set size 1,1
set multiplot
unset key

# first
set size 0.5,0.5
set origin 0.0,0.5
set xtics 0,25
set xlabel 'x'
set ylabel 'y'
set title 'first'
plot 'data.txt' using 1:2 w l lw 0.5 lc rgb 'red'
set xtics auto

# second
set origin 0.5,0.5
set xlabel 'x1'
set ylabel 'y1'
set title 'second'
plot 'data1.txt' using 1:2 w l lw 1 lc rgb 'blue'

# third
set size 1,0.5
set origin 0,0
set view 60,15
set ytics 0,10
set ztics 0,75
set title 'third'
set xlabel 'x2'
set ylabel 'y2' offset -10,0
set zlabel 'z2'
splot 'data.txt' using 1:2:3 w l lw 2.5 lc rgb 'gray'
set ytics auto
set ztics auto

unset multiplot
